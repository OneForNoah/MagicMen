Create Table users (
            id_user INT NOT NULL,
            name VARCHAR( 128 ) NOT NULL,
            email VARCHAR( 64 ) NOT NULL,
            phone_number VARCHAR( 16 ) NOT NULL,
            username VARCHAR( 16 ) NOT NULL,
            password VARCHAR( 32 ) NOT NULL,
            PRIMARY KEY ( id_user )
            );
